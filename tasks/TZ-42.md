ШАГ 4: Создать inline static file server для раздачи frontend bundle

  Под-шаг 4.1: Новая функция `serveStatic(rootDir, port)` — НИКАКИХ новых npm-deps.
  Под-шаг 4.2: Использует `node:http` + `node:fs` (createReadStream, statSync).
  Под-шаг 4.3: MIME map для .html, .js, .css, .json, .svg, .png, .jpg, .ico, .woff2, .woff, .ttf.
  Под-шаг 4.4: SPA fallback: если файл не найден, отдаёт `index.html` (для Angular Router).
  Под-шаг 4.5: Path traversal protection: `if (!filePath.startsWith(rootDir))` → 403.
  Под-шаг 4.6: Cache headers: `Cache-Control: public, max-age=31536000, immutable` для
               `/assets/*` (Angular build кладут hashed assets в `/assets/`).
  Под-шаг 4.7: `index.html` — `Cache-Control: no-cache` (для обновления при редеплое).
  Под-шаг 4.8: Returns { server, port } — server.listen() происходит в main().
  Под-шаг 4.9: Опционально: gzip через `zlib.createGzip()` — ОТЛОЖЕНО (не блокер).

  Пример структуры:
  ```js
  function serveStatic(rootDir, port) {
    const MIME = { '.html': 'text/html; charset=utf-8', /* ... */ };
    const server = createServer((req, res) => {
      const url = new URL(req.url, `http://${req.headers.host}`);
      let filePath = normalize(join(rootDir, url.pathname));
      if (!filePath.startsWith(normalize(rootDir))) { res.writeHead(403); res.end('forbidden'); return; }
      try {
        const stat = statSync(filePath);
        if (stat.isDirectory()) { /* fallthrough to SPA */ throw new Error('is dir'); }
        const mime = MIME[extname(filePath).toLowerCase()] || 'application/octet-stream';
        const isHashedAsset = url.pathname.startsWith('/assets/');
        res.writeHead(200, {
          'Content-Type': mime,
          'Content-Length': stat.size,
          ...(isHashedAsset ? { 'Cache-Control': 'public, max-age=31536000, immutable' } : {}),
        });
        createReadStream(filePath).pipe(res);
      } catch {
        // SPA fallback
        try {
          const idx = readFileSync(join(rootDir, 'index.html'));
          res.writeHead(200, { 'Content-Type': 'text/html; charset=utf-8', 'Cache-Control': 'no-cache' });
          res.end(idx);
        } catch { res.writeHead(404); res.end('not found'); }
      }
    });
    server.listen(port, '0.0.0.0');
    return server;
  }
  ```

13. `node start.mjs --prod --reset` (если оба переданы) — fail fast с понятной

15. Cold start (build + boot): ~60-120s. Warm start (dist/ уже есть + пересборка инкрементальная): ~30-60s.
16. Frontend bundle size показывается в printReadyPanel (recursive size of dist/frontend/browser/).

          if (entry.isDirectory()) walk(p);
          else total += statSync(p).size;
        }
      };
      walk(dir);
      return total;
    }
