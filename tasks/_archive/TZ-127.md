**ШАГ 1: Tiered rate-limit guard — `tiered-throttler.guard.ts` (NEW)**

1.1. Создать файл `backend/src/common/guards/tiered-throttler.guard.ts`:
```ts
import { Injectable, ExecutionContext } from '@nestjs/common';
import { ThrottlerGuard } from '@nestjs/throttler';

const TIERS = {
  anonymous: { ttl: 60000, limit: 20 },   // 20 RPM
  user:      { ttl: 60000, limit: 300 },  // 300 RPM (5 req/s avg)
  admin:     { ttl: 60000, limit: 1500 }, // 1500 RPM (25 req/s avg)
};

@Injectable()
export class TieredThrottlerGuard extends ThrottlerGuard {
  protected async getTracker(req: any): Promise<string> {
    const xff = req.headers?.['x-forwarded-for'];
    const ip = Array.isArray(xff) ? xff[0] : xff ?? req.ip ?? 'unknown';
    const userSub = req.user?.sub ?? 'anon';
    return `user:${userSub}:ip:${ip}`;
  }

  protected async shouldSkip(_ctx: ExecutionContext): Promise<boolean> {
    if (process.env.DISABLE_THROTTLE === '1') return true;
    return false;
  }

  // Tier detection via @nestjs/throttler v6 canonical pattern: override
  // `getThrottlerSuffix` only — base class uses it in `generateKey` for
  // storage-isolated namespace per tier (no manual throttler splicing).
  // ThrottlerModule.forRootAsync registration lives in app.module.ts (ШАГ 1.3),
  // providing three named throttler entries ('anonymous' / 'user' / 'admin').
  protected getThrottlerSuffix(req: Record<string, unknown>): Promise<string> {
    const u = (req as { user?: { role?: string } }).user;
    return Promise.resolve(
      u?.role === 'admin' ? 'admin'
        : u?.role ? 'user'
        : 'anonymous',
    );
  }
}
```

1.2. Удалить `throttler-behind-auth.guard.ts` (legacy bypass) → заменить на `tiered-throttler.guard.ts`.

1.3. В `backend/src/app.module.ts` заменить текущий `ThrottlerModule.forRoot([...])` (lines 88-93) на `ThrottlerModule.forRootAsync`:

```ts
import { ThrottlerModule, ThrottlerStorageService } from '@nestjs/throttler';
import { APP_GUARD } from '@nestjs/core';
import { TieredThrottlerGuard } from './common/guards/tiered-throttler.guard';

// В @Module({ imports: [...] }):
ThrottlerModule.forRootAsync({
  useFactory: () => ({
    throttlers: [
      { name: 'anonymous', ttl: 60_000, limit: 20 },   // 20 req/min anon
      { name: 'user',      ttl: 60_000, limit: 300 },  // 5 req/s avg user
      { name: 'admin',     ttl: 60_000, limit: 1500 }, // 25 req/s avg admin
    ],
    // Storage service is in-memory by default; for multi-instance deploy
    // switch to `@nestjs/throttler-storage-redis` + property env REDIS_URL.

}),

// В @Module({ providers: [...] }):
```

- Current `ThrottlerModule.forRoot([{ name: 'short', ttl: 1000, limit: 10 }, { name: 'long', ttl: 60000, limit: 100 }])` имеет 2 throttlers с другими names. Удалить эти 2 entries целиком — они теряются при switching на tiered model.

**ШАГ 2: Refresh token в HttpOnly cookie — `cookie-parser` middleware**

2.2. В `backend/src/main.ts`:
```ts
import cookieParser from 'cookie-parser';
// `secret` enables SIGNED cookies (HMAC) — protects against network tampering
// of HttpOnly cookies (XSS protection alone is NOT sufficient if HTTPS is
// misconfigured / pinned incorrectly).
// Production: REQUIRED to set COOKIE_SECRET in .env (>=32 random hex chars).
const cookieSecret = process.env.COOKIE_SECRET ?? 'kppdf-development-cookie-secret-CHANGE-IN-PROD';
if (process.env.NODE_ENV === 'production') {
  const isWeak =
    !process.env.COOKIE_SECRET ||
    process.env.COOKIE_SECRET.length < 32 ||
    process.env.COOKIE_SECRET.includes('change-me') ||
    process.env.COOKIE_SECRET.includes('development');
  if (isWeak) {
    throw new Error(
      'COOKIE_SECRET must be set in production (>=32 hex chars, must NOT contain "change-me" or "development"). Refusing to start.',
    );
  }
}
app.use(cookieParser(cookieSecret));   // перед guards
```
